                fun end_time ()
                    =
                    {   (since() ++++ *this)
                            ->
                            x as { usr, sys, gc };

                        this  :=  zeros;
                        cumulative :=  *cumulative ++++ x;

                        usr +++ sys +++ gc;
                    };

                fun finish ()
                    =
                    {   current :=  prev;
                        #
                        if *say_end
                            #
                            time = time_to_string (end_time ());
                            #
                            if (time != "0.00" or *say_when_nonzero)
                                #
                                say "End   ";
                                sayfield (40, name);

                                if *keep_time   apply say ["    ", time, " sec\n"];
                                else            say "\n";
                                fi;

                                flush();
                            fi;
                        else
                            end_time();
                            ();
                        fi;
                    };

                if *keep_time
                    #
                    t' :=  since() ++++ *t';
                fi;

                current :=  p;

                if *say_begin
                    #
                    apply say ["Begin ", name, "\n"];
                    flush();
                fi;

                (   (f x)
                    except
                        e =  {   finish ();
                                 raise exception e;
                             }
                )
                then
                    finish ();
            };
