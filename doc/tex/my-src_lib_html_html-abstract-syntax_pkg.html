    fun match sl s
        =
        {
            compare = string::compare_sequences
                  (\\ (c1, c2) => char::compare (char::to_upper c1, c2); end );

            fun eq (REF s')
                =
                (compare (s, s') == exceptions::EQUAL);
        
            list::find eq sl;
        };
