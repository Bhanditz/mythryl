                =

                            # let denseBytes  = (max_codetemp_id * (max_codetemp_id + 1) + 15) div 16
                            # in  GRAPH_BY_EDGE_HASHTABLE (rw_vector_of_one_byte_unts::rw_vector (denseBytes, 0u0))
                            # end 
                            # else

                            (round_size  hashchains_count_hint)
                                ->

                                            #

                                            #

                                            uses                =>  REF [],

                                            #   
                                            priority            =>  REF 0.0,
                                            movecost            =>  REF 0.0,
                                            #
