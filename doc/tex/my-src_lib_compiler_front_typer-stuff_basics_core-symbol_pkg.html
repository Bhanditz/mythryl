# The internal symbol that is used to find "core" namings.  This symbol
# is "package _Core", which is something that cannot legally occur in
# normal Mythryl code, so there is no danger of accidentially overriding
# this naming.

package core_symbol {
    #
    core_symbol = symbol::make_package_symbol "_Core";
};
