api Cmemalloc {

    exception OUT_OF_MEMORY;

    eqtype Addr';               #  To avoid clash with address from CMEMACCESS 

    allot:  Unt -> Addr';       #  May raise OUT_OF_MEMORY 
    free:   Addr' -> Void;
};
