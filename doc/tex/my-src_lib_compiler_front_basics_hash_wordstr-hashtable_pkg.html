                                                                        # typelocked_hashtable_g        is from   src/lib/src/typelocked-hashtable-g.pkg
package word_string_hashtable
    =
    typelocked_hashtable_g (

        Hash_Key = (Unt, String);

        fun hash_value (k:  Hash_Key)
            =
            #1 k;

        fun same_key (  (h,  s ):   Hash_Key,
                        (h', s')
                     )
            =
            h == h'   and
            s == s';
    );
