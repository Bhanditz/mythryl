                        fun min (a: Int, b) =  (a < b  ??  a  ::  b);
                        fun max (a: Int, b) =  (a > b  ??  a  ::  b);

                        fun union (row, x1, x2)
                            = 
                            rw_vector::set
                              ( damage,
                                row,
                                ins (rw_vector::get (damage, row))
