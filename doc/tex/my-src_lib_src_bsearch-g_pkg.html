    # binary search on ordered typelocked arrays. The comparison function
    # compare embeds a projection function from the element type to the key
    # type.

    fun bsearch compare (key, arr)
        =
        get (0, a::length arr - 1)
        where
            fun get (lo, hi)
                = 
                if   (hi >= lo)

                     m = lo + (hi - lo) / 2;
                     x = a::get (arr, m);
                  
                     case (compare (key, x))
                          LESS    =>  get (lo, m - 1);
                          EQUAL   =>  (THE (m, x));
                          GREATER =>  get (m+1, hi);
                     esac;
                  
                else
                     NULL;
                fi;
          
        end;

}; #  BSearch
