package burg_ast {


     Decl_Ast = START  String
                      | TERM   List ((String, Null_Or( String )) )
                      | TERMPREFIX  String
                      | RULEPREFIX  String
                      | BEGIN_API  String;

     Pattern_Ast = PAT  ((String, List( Pattern_Ast )) );

     Rule_Ast = RULE  ((String, Pattern_Ast, String, List Int));

     Spec_Ast = SPEC  { head:  List( String ),
                                 decls:  List( Decl_Ast ), 
                                 rules:  List( Rule_Ast ),
                                 tail:  List( String ) };
}; #  BurgAST
