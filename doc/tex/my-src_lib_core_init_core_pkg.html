            fun onestage sel ()
                =
                {   fns = map sel *hook;

                    \\ arg =  apply  (runwith arg)  fns;
                };

            fun twostage sel ()
                =
                {   stage1_fns = map sel *hook;

                    \\ arg
                        =
                        {   stage2_fns = revmap (runwith arg) stage1_fns;

                            \\ () =  apply (runwith ()) stage2_fns;
                        };
                };

        herein

            fun tdp_reserve n
                =
                {   r = *next;
                    next := r + n;
                    r;
                };

            fun tdp_reset ()
                =
                next := 0;
