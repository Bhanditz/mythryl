                fun point_to_value (point: g2d::Point)
                    =
                    {   gutter_box  ->  { row, col, high, wide };
                        #
                        high         =  int::max (high, 1);                                                                     # Prevent divide-by-zero;

                        fpixels      =  float::from_int high;
                        fvalues      =  upper_limit - lower_limit;

                        p_to_v       =  fvalues / fpixels;

                        value        =  float::from_int (row + high - point.row)  *  p_to_v;

                        value        =  float::min (value, upper_limit);
                        value        =  float::max (value, lower_limit);

                        value;
                    };

                fun thumb_displaylist { lower_limit, slider_value, upper_limit, gutter_box, coverage }                          # Thumb shows portion of file currently visible in window. If coverage==1.0, all the file is visible and thumb fills gutter.  If coverage==0.5, half the file is visible, and thumb fills half of gutter.
                    =                                                                                                           # Position of thumb shows which part of file is visible: Top, middle, bottom, whatever.
                    {   gutter_box  ->  { row, col, high, wide };
                        #
                        thumb_height =  float::round ((float::from_int high) * coverage);                                       # Pixel height of thumb.
                        thumb_range  =  (float::from_int high) * (1.0 - coverage);                                              # Number of pixels which thumb is free to move.
                        value_range  =   upper_limit - lower_limit;                                                             # Number of values which slider_value is free to range over.
                        fvalue       =   slider_value - lower_limit;                                                            # Zero-based value of slider_value.
                        v_to_p       =  thumb_range / value_range;                                                              # Conversion factor from slider_value range to thumb range.
                        thumb_lo     =  row + high - (float::round (fvalue * v_to_p));
                        thumb_hi     =  thumb_lo - thumb_height;

                        thumb_box    =  { col => col + 2, row => thumb_hi, wide => wide - 4, high => thumb_height };                    # 

                        thumb_body   = [ gd::COLOR ( rgb::black, [ gd::FILLED_BOXES [ thumb_box ]]) ];

                        thumb_body;
                    };
                    
                fun cursor_displaylist { lower_limit, slider_value, upper_limit, gutter_box }
                    =
                    {   gutter_box  ->  { row, col, high, wide };
                        #
                        fpixels      =  float::from_int high;
                        fvalues      =  upper_limit - lower_limit;
                        fvalue       =  slider_value;

                        v_to_p       =  fpixels / fvalues;

                        cursor_mid   =  row  + high - (float::round  (fvalue * v_to_p));

                        cursor_high2  =  10;                                                                                    # Half-height of cursor.
                        cursor_height =  2*cursor_high2 + 1;

                        cursor_row   =  cursor_mid - cursor_high2;

                        cursor_box   =  { col => col + 4, row => cursor_row, wide => wide - 8, high => cursor_height };         # "+ 4" and "- 8" so the cursor outline is cleanly separated from the gutter frame.

                        (g2d::box::box_corners  cursor_box)
                            ->
                            { upper_left, lower_left, lower_right, upper_right };

                        left_mid  = g2d::point::mean [ upper_left,  lower_left  ];
                        right_mid = g2d::point::mean [ upper_right, lower_right ];

                        cursor_outline = [ left_mid, right_mid, upper_right, upper_left, lower_left, lower_right, right_mid ];

                        [ gd::COLOR ( rgb::white, [ gd::FILLED_BOXES [ cursor_box ]]) ]
                        @
                        [ gd::COLOR ( rgb::rgb_mix01(0.9,rgb::black,rgb::white), [ gd::LINE_THICKNESS (0, [ gd::PATH cursor_outline ]) ]) ];
                    };
                    

                foreground =    [ gd::COLOR (palette.body_color, [ gd::FILLED_POLYGON (g2d::box::to_points inner_box) ]) ];                                     # Interior of gutter. We draw this first because 3D outline occupies same bounding box:

                foreground =    if (coverage == 0.0)   foreground;
                                else                   foreground @ thumb_displaylist { lower_limit, slider_value, upper_limit, gutter_box, coverage };
                                fi;

                foreground =    foreground  @  cursor_displaylist { lower_limit, slider_value, upper_limit, gutter_box };                                       # Draw cursor next because we want it to overwrite gutter interior but be overwritten by gutter frame.

                                    hitext  =   sprintf "%.3g" upper_limit;

                                    hidims  =   get_text_dimensions  hitext;

                                    mipoint =   g2d::box::midpoint inner_box;

                                    textcol =   mipoint.col;
                                    textrow =   mipoint.row - lodims.font_descent + ((lodims.font_ascent + lodims.font_descent) / 2); 

                                    hipoint =   { col => textcol,  row => inner_box.row                  + (lodims.font_ascent                        + 10) };
                                    mipoint1=   { col => textcol,  row =>   mipoint.row                  - (                      lodims.font_descent -  2) };
                                    mipoint2=   { col => textcol,  row =>   mipoint.row                  + (lodims.font_height2 - lodims.font_descent +  2) };
                                    mipoint3=   { col => textcol,  row =>   mipoint.row                  + (lodims.font_ascent                        +  0) };
                                    lopoint =   { col => textcol,  row => inner_box.row + inner_box.high - (                      lodims.font_descent + 10) };


                                    lodraw  =   [ gd::PUT_TEXT   ( gd::CENTERED_ON_POINT,
                                                                   [ gd::TEXT (lopoint, lotext) ]
                                                                 )
                                                ];      

                                    midraw  =   case (text, show_value)
                                                    #

                                                                                         )
                                                                        ];
                                                    (THE t, FALSE) =>   [ gd::PUT_TEXT   ( gd::CENTERED_ON_POINT,
                                                                                           [ gd::TEXT (mipoint2, t     ) ]
                                                                                         )
                                                                        ];
                                                    (THE t, TRUE ) =>   [ gd::PUT_TEXT   ( gd::CENTERED_ON_POINT,
                                                                                           [ gd::TEXT (mipoint1, t     ) ]
                                                                                         ),
                                                                          gd::PUT_TEXT   ( gd::CENTERED_ON_POINT,
                                                                                           [ gd::TEXT (mipoint3, mitext) ]
                                                                                         )
                                                                        ];
                                                esac;


                                    hidraw  =   [ gd::PUT_TEXT   ( gd::CENTERED_ON_POINT,
                                                                   [ gd::TEXT (hipoint, hitext) ]
                                                                 )
                                                ];      


                                    display_list =  if show_limits   lodraw @ midraw @ hidraw;
                                                    else                      midraw;
                                                    fi; 



                                    display_list =  case display_list   [] =>  [];
