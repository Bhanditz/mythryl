## threadkit-debug.api
#
# Debugging support for the threadkit core.

# Compiled by:
#     src/lib/std/standard.lib

# This api is implemented in:
#
#     src/lib/src/lib/thread-kit/src/core-thread-kit/threadkit-debug.pkg
#
api Threadkit_Debug {
    #
    say_debug:          String -> Void;
    say_debug_ts:       String -> Void;                                 # "_ts" may be (with) "_timestamp".
    say_debug_id:       String -> Void;                                 # "_id" may be (with) "_process_id" (i.e., linux pid).
};
