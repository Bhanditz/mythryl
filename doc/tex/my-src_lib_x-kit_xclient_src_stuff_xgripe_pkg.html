stipulate
    package f = file__premicrothread;
herein

    package xgripe {

        exception XERROR  String;

        fun impossible string =  raise exception (XERROR string);
        fun xerror     string =  raise exception (XERROR string);

        no_warning = REF FALSE;

        fun warning s
            =
           if *no_warning  ();
           else            f::write (f::stderr, s);
           fi;
    };

end;
