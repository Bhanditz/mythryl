                                         _   => {   emit "fldl\t";

                                     };
                mcf::FLDT operand => {   emit "fldt\t";

                                     };
                mcf::FILD operand => {   emit "fild\t";

                mcf::FILDL operand => {   emit "fildl\t";

                mcf::FILDLL operand => {   emit "fildll\t"; 
                                           put_operand operand; 
                                       };

                mcf::FENV { fenv_op,

                          }

                mcf::FMOVE { fsize,

                mcf::FIBINOP { isize,

                    => {   if  i

                               emit "fcmp";

                           {   put_fsize fsize; 
                               emit "\t"; 
                               put_lsrc lsrc; 
                               emit ", "; 
                               put_rsrc rsrc; 
                           }; 
                       };
                mcf::SAHF => emit "sahf";
                mcf::LAHF => emit "lahf";
                mcf::SOURCE { } => emit "source";
                mcf::SINK { } => emit "sink";
                mcf::PHI { } => emit "phi";
            esac;

                        =>
